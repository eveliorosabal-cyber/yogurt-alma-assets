        // ========== CARGAR PRODUCTOS CON VERIFICACIÓN DE IMAGEN ==========
        async function loadProductsFromJSON() {
            const base = `https://raw.githubusercontent.com/${CONFIG_GITHUB.USER}/${CONFIG_GITHUB.REPO}/${CONFIG_GITHUB.BRANCH}/`;
            const url = base + 'productos.json?t=' + Date.now();
            try {
                const resp = await fetch(url);
                if (!resp.ok) throw new Error(`HTTP ${resp.status}`);
                const productos = await resp.json();

                const loader = document.createElement('div');
                loader.id = 'img-verifier';
                loader.innerHTML = '<div style="position:fixed; top:50%; left:50%; transform:translate(-50%,-50%); background:var(--color-primary); color:white; padding:1rem 2rem; border-radius:40px; z-index:10000; font-weight:bold;">Verificando imágenes...</div>';
                document.body.appendChild(loader);

                const verificarImagen = async (producto) => {
                    const folderKey = CONFIG.categories.folders[producto.category];
                    if (!folderKey) return null;
                    const imageUrl = `${base}images/productos del dia/${folderKey}/${encodeURIComponent(producto.imageFile)}`;
                    try {
                        const headResp = await fetch(imageUrl, { method: 'HEAD' });
                        if (headResp.ok) {
                            producto.image = imageUrl;
                            return producto;
                        } else {
                            console.warn(`Imagen no encontrada: ${imageUrl} - omitido: ${producto.name}`);
                            return null;
                        }
                    } catch (err) {
                        console.warn(`Error verificando ${imageUrl}:`, err);
                        return null;
                    }
                };

                const resultados = await Promise.all(productos.map(p => verificarImagen(p)));
                const productosValidos = resultados.filter(p => p !== null);

                CATALOG = {};
                for (const p of productosValidos) {
                    const cat = p.category;
                    if (!CATALOG[cat]) CATALOG[cat] = [];
                    CATALOG[cat].push(p);
                }

                console.log('Productos válidos (con imagen real):', CATALOG);
                const loaderEl = document.getElementById('img-verifier');
                if (loaderEl) loaderEl.remove();

                if (productosValidos.length === 0) {
                    categoriesContainer.innerHTML = '<div class="error-message">⚠️ No hay productos disponibles hoy. Revisa más tarde.</div>';
                }
            } catch (e) {
                console.error(e);
                alert('Error al cargar productos.');
                throw e;
            }
        }

        let sliderInterval, currentSlide = 0, sliderItems = [];
        async function initSlider() {
            const container = document.getElementById('hero-slider');
            const loading = document.getElementById('slider-loading');
            const dotsContainer = document.getElementById('slider-dots');
            if (!container) return;

            const base = `https://raw.githubusercontent.com/${CONFIG_GITHUB.USER}/${CONFIG_GITHUB.REPO}/${CONFIG_GITHUB.BRANCH}/`;
            const folderPath = CONFIG_GITHUB.PATHS.slider;
            const url = `https://api.github.com/repos/${CONFIG_GITHUB.USER}/${CONFIG_GITHUB.REPO}/contents/${encodeURIComponent(folderPath)}?ref=${CONFIG_GITHUB.BRANCH}`;

            try {
                const resp = await fetch(url);
                if (!resp.ok) throw new Error();
                const files = await resp.json();
                const mediaFiles = files.filter(f => /\.(jpg|jpeg|png|gif|mp4|webm|mov)$/i.test(f.name));
                if (mediaFiles.length === 0) throw new Error();

                sliderItems = mediaFiles.map(f => {
                    const isVideo = /\.(mp4|webm|mov)$/i.test(f.name);
                    return { type: isVideo ? 'video' : 'image', url: f.download_url };
                });

                while (container.firstChild) container.removeChild(container.firstChild);
                sliderItems.forEach((item, idx) => {
                    const slideDiv = document.createElement('div');
                    slideDiv.className = `slide ${idx === 0 ? 'active' : ''}`;
                    if (item.type === 'image') {
                        slideDiv.style.backgroundImage = `url('${item.url}')`;
                        slideDiv.style.backgroundSize = 'cover';
                        slideDiv.style.backgroundPosition = 'center';
                    } else {
                        const video = document.createElement('video');
                        video.src = item.url;
                        video.autoplay = true;
                        video.muted = true;
                        video.loop = true;
                        video.playsInline = true;
                        video.style.width = '100%';
                        video.style.height = '100%';
                        video.style.objectFit = 'cover';
                        slideDiv.appendChild(video);
                    }
                    container.appendChild(slideDiv);
                });

                dotsContainer.innerHTML = '';
                sliderItems.forEach((_, i) => {
                    const dot = document.createElement('span');
                    dot.className = `dot ${i === 0 ? 'active' : ''}`;
                    dot.onclick = () => goToSlide(i);
                    dotsContainer.appendChild(dot);
                });

                loading.style.display = 'none';
                startSliderAuto();
            } catch (e) {
                container.innerHTML = `<div class="slide active" style="background-image: url('https://via.placeholder.com/1920x1080/2C1810/FAF7F2?text=Yogurt+Alma');"></div>`;
                loading.style.display = 'none';
            }
        }
        function startSliderAuto() {
            if (sliderInterval) clearInterval(sliderInterval);
            sliderInterval = setInterval(() => {
                const slides = document.querySelectorAll('.slide');
                const dots = document.querySelectorAll('.dot');
                if (slides.length === 0) return;
                slides[currentSlide].classList.remove('active');
                dots[currentSlide].classList.remove('active');
                currentSlide = (currentSlide + 1) % slides.length;
                slides[currentSlide].classList.add('active');
                dots[currentSlide].classList.add('active');
            }, 5000);
        }
        function goToSlide(index) {
            const slides = document.querySelectorAll('.slide');
            const dots = document.querySelectorAll('.dot');
            if (slides.length === 0) return;
            slides[currentSlide].classList.remove('active');
            dots[currentSlide].classList.remove('active');
            currentSlide = index;
            slides[currentSlide].classList.add('active');
            dots[currentSlide].classList.add('active');
            clearInterval(sliderInterval);
            startSliderAuto();
        }

        function updateQuantity(productId, delta) {
            const newQty = (cart[productId] || 0) + delta;
            if (newQty >= 0) {
                cart[productId] = newQty;
                const el = document.getElementById(`qty-${productId}`);
                if (el) el.textContent = newQty;
                updateCart();
            }
        }

        function updateCart() {
            const allProducts = Object.values(CATALOG).flat();
            const items = allProducts.filter(p => cart[p.id] > 0);
            const total = items.reduce((sum, p) => sum + (cart[p.id] * p.price), 0);
            document.getElementById('cart-total').textContent = total.toLocaleString('es-PY');
            document.getElementById('cart-float').classList.toggle('active', items.length > 0);
            if (modalBody && modalBody.children.length) updateCheckoutSummary(total);
        }

            }
        }

        function selectPayment(method) {
            currentPayment = method;
            document.querySelectorAll('.payment-option').forEach(opt => opt.classList.remove('selected'));
            document.getElementById(`pay-${method}`).classList.add('selected');

        function calculateChange() {
            const amount = parseInt(document.getElementById('cash-amount').value) || 0;
            const allProducts = Object.values(CATALOG).flat();
            const total = allProducts.reduce((sum, p) => sum + (cart[p.id] * p.price), 0);
            const change = amount - total;
            const display = document.getElementById('change-display');
            if (!display) return;
            if (amount === 0) { display.style.display = 'none'; return; }
            display.style.display = 'block';
            if (change < 0) {
                display.className = 'vuelto-box error';
                display.textContent = `Faltan Gs. ${Math.abs(change).toLocaleString('es-PY')}`;
            } else {
                display.className = 'vuelto-box ok';
                display.textContent = `Vuelto: Gs. ${change.toLocaleString('es-PY')}`;
            }
        }

        }

        async function confirmOrder() {
            const name = document.getElementById('client-name').value.trim();
            const phone = document.getElementById('client-phone').value.trim();
            if (!name) { alert('Ingresá tu nombre'); return; }
            if (!phone) { alert('Ingresá tu WhatsApp'); return; }

            if (currentPayment === 'cash') {
                const amount = parseInt(document.getElementById('cash-amount').value) || 0;
                const allProducts = Object.values(CATALOG).flat();
                const total = allProducts.reduce((sum, p) => sum + (cart[p.id] * p.price), 0);
                if (amount < total) { alert('El monto es insuficiente'); return; }
            }

            document.getElementById('loading-screen').classList.add('active');

            const allProducts = Object.values(CATALOG).flat();
            const items = allProducts.filter(p => cart[p.id] > 0);
            const subtotal = items.reduce((sum, p) => sum + (cart[p.id] * p.price), 0);
            const order = {
                id: 'YA-' + Date.now().toString().slice(-6),
                timestamp: new Date().toISOString(),
                customer: { name, phone },
                items: items.map(p => ({ name: p.name, qty: cart[p.id], price: p.price })),
                totals: { subtotal, total: subtotal },
                delivery: { type: 'pickup', address: 'Retiro en local - ' + CONFIG.store.name },
                payment: { method: currentPayment, alias: currentPayment === 'transfer' ? CONFIG.store.alias : null }
            };

            const message = generateSecureMessage(order);
            const whatsappUrl = `https://wa.me/${CONFIG.store.whatsapp}?text=${encodeURIComponent(message)}`;

            try {
                await fetch(CONFIG.store.google_sheets_url, {
                    method: 'POST',
                    mode: 'no-cors',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(order)
                });
            } catch (e) {}

            setTimeout(() => {
                document.getElementById('loading-screen').classList.remove('active');
                const successScreen = document.getElementById('success-screen');
                successScreen.classList.add('active');
                document.getElementById('whatsapp-btn').href = whatsappUrl;
                document.getElementById('success-message').textContent = currentPayment === 'transfer'
                    ? `Tu pedido está pendiente de pago.\n\nAlias: ${CONFIG.store.alias}\nMonto: Gs. ${order.totals.total.toLocaleString('es-PY')}\n\nUna vez realizada la transferencia, envianos el comprobante por WhatsApp.`
                    : 'Tu pedido está confirmado. Podés pasar a retirar por nuestro local. Te esperamos.';
            }, 1500);
        }

        function generateSecureMessage(order) {
            const lines = [
                '🥛 *YOGURT ALMA - PEDIDO OFICIAL*',
                '',
                `*N° Pedido:* ${order.id}`,
                `*Fecha:* ${new Date().toLocaleString('es-PY')}`,
                '',
                '*CLIENTE:*',
                `Nombre: ${order.customer.name}`,
                `Tel: ${order.customer.phone}`,
                '',
                '*PRODUCTOS:*'
            ];
            order.items.forEach(item => {
                lines.push(`• ${item.qty}x ${item.name}`);
                lines.push(`  Gs. ${(item.qty * item.price).toLocaleString('es-PY')}`);
            });
            lines.push('');
            lines.push('*TOTAL:*');
            lines.push(`*Gs. ${order.totals.total.toLocaleString('es-PY')}*`);
            lines.push('');
            lines.push('*ENTREGA:*');
            lines.push(`Tipo: Retiro en local`);
            lines.push(`Dirección: ${order.delivery.address}`);
            lines.push('');
            lines.push('*PAGO:*');
            if (order.payment.method === 'transfer') {
                lines.push(`Método: Transferencia`);
                lines.push(`Alias: ${order.payment.alias}`);
                lines.push(`Estado: PENDIENTE (esperando comprobante)`);
            } else {
                lines.push(`Método: Efectivo`);
                const amount = document.getElementById('cash-amount').value;
                const change = parseInt(amount) - order.totals.total;
                lines.push(`Paga con: Gs. ${parseInt(amount).toLocaleString('es-PY')}`);
                lines.push(`Vuelto: Gs. ${change.toLocaleString('es-PY')}`);
            }
            lines.push('');
            lines.push('✅ Pedido generado automáticamente');
            lines.push('📍 Por favor no modifiques este mensaje');
            return lines.join('\n');
        }

        function initScrollAnimations() {
            const observer = new IntersectionObserver((entries) => {
                entries.forEach(e => { if (e.isIntersecting) e.target.classList.add('visible'); });
            }, { threshold: 0.1 });
            document.querySelectorAll('.fade-up').forEach(el => observer.observe(el));
        }

            });
        }

        function downloadJSONTemplate() {
            const allProducts = Object.values(CATALOG).flat();
            const template = allProducts.map(p => ({
                id: p.id, name: p.name, desc: p.desc, weight: p.weight,
                price: p.price, category: p.category, badge: p.badge || null, imageFile: p.imageFile
            }));
            const blob = new Blob([JSON.stringify(template, null, 2)], { type: 'application/json' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = 'productos.json';
            a.click();
            URL.revokeObjectURL(url);
        }

                localStorage.setItem('yogurt-theme', theme);
            }));
